Cvičení: Prohození proměnných

// Tvým úkolem je pomocí JavaScriptu prohodit hodnoty z proměnné a do b.
// Zde začíná funkce, která ověří, jestli se ti zadání povedlo splnit.
// S touto částí nemůžeš manipulovat

function Test(a, b) {
if (a == 12) && (b == 10){
return alert("Bravo!");
}
else
return alert("a = " + a + "\\n" + "b = " + b + "\\nNěco není správně!");
}

// Zde začíná část, kde se deklarují proměnné a, b.
// S touto částí také nemůžeš manipulovat.

let a = "10";
let b = "12";

// Po tvém kódu přichází příkaz, který spustí test, jestli jsi zapsal kód správně
Test(a, b);
